import pandas as pd
import matplotlib.pyplot as plt1
import matplotlib.pyplot as plt2

df1=pd.read_csv('beneficiarios_primera_casa_BA-2018.csv')

x_values_edad = df1['edad'].unique()
y_values_edad = df1['edad'].value_counts().tolist()
y_values_edad.reverse()

#grafico1
plt1.bar(x_values_edad,y_values_edad)
plt1.title('CUANTAS EDADES HAY DE CADA UNA')
plt1.xlabel('PERSONAS DE DIFERENTES EDADES')
plt1.ylabel('CANTIDAD DE CADA EDAD')
plt1.show()

#____________________________________________________________________________
x_values_alquila = df1['alquila'].unique()
y_values_alquila = df1['alquila'].value_counts().tolist()
y_values_alquila.reverse()

#grafico2
plt2.bar(x_values_alquila,y_values_alquila)
plt2.title('CUANTAS PERSONAS ALQUILAN')
plt2.xlabel('ALQUILA')
plt2.ylabel('CANTIDAD QUE ALQUILA')
plt2.show()
